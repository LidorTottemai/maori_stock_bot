## 1 — Metadata (Next.js 14+ App Router)

```ts
// app/[locale]/layout.tsx
import { Metadata } from "next"

const business = {
  name: process.env.NEXT_PUBLIC_BUSINESS_NAME!,
  description: process.env.NEXT_PUBLIC_BUSINESS_DESCRIPTION!,
  phone: process.env.NEXT_PUBLIC_PHONE!,
  city: process.env.NEXT_PUBLIC_CITY!,
  domain: process.env.NEXT_PUBLIC_DOMAIN!,
}

export async function generateMetadata({ params }: { params: { locale: string } }): Promise<Metadata> {
  const locale = params.locale
  const isHebrew = locale === "he"
  
  return {
    title: {
      template: `%s | ${business.name}`,
      default: business.name,
    },
    description: business.description,
    
    openGraph: {
      type: "website",
      siteName: business.name,
      locale: isHebrew ? "he_IL" : "en_US",
      alternateLocale: [isHebrew ? "en_US" : "he_IL"],
      url: `https://${business.domain}/${locale}`,
      images: [{ url: `/og-image.png`, width: 1200, height: 630 }],
    },
    
    twitter: {
      card: "summary_large_image",
      title: business.name,
      description: business.description,
      images: ["/og-image.png"],
    },
    
    robots: { index: true, follow: true, googleBot: { index: true, follow: true } },
    
    alternates: {
      canonical: `https://${business.domain}/${locale}`,
      languages: {
        "he": `https://${business.domain}/he`,
        "en": `https://${business.domain}/en`,
        "x-default": `https://${business.domain}/he`,
      },
    },
    
    verification: {
      google: process.env.GOOGLE_SITE_VERIFICATION,
    },
  }
}
```

## 2 — Structured Data: LocalBusiness

```tsx
// components/seo/LocalBusinessSchema.tsx
interface LocalBusinessSchemaProps {
  name: string
  description: string
  phone: string
  email?: string
  address: { street: string; city: string; zip?: string }
  hours?: string[]    // ["Mo-Fr 09:00-18:00", "Sa 10:00-14:00"]
  priceRange?: string  // "₪₪"
  logo?: string
  images?: string[]
  category?: string    // "HealthAndBeautyBusiness" | "Restaurant" | "LocalBusiness"
}

export function LocalBusinessSchema({
  name, description, phone, email, address, hours, priceRange, logo, images, category = "LocalBusiness"
}: LocalBusinessSchemaProps) {
  const domain = process.env.NEXT_PUBLIC_DOMAIN!
  
  const schema = {
    "@context": "https://schema.org",
    "@type": category,
    name,
    description,
    telephone: phone,
    email,
    url: `https://${domain}`,
    logo: logo ? `https://${domain}${logo}` : undefined,
    image: images?.map(img => `https://${domain}${img}`),
    address: {
      "@type": "PostalAddress",
      streetAddress: address.street,
      addressLocality: address.city,
      addressCountry: "IL",
      postalCode: address.zip,
    },
    openingHoursSpecification: hours?.map(h => ({ "@type": "OpeningHoursSpecification", dayOfWeek: h })),
    priceRange,
    areaServed: {
      "@type": "City",
      name: address.city,
    },
    inLanguage: ["he", "en"],
  }

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(schema, null, 2) }}
    />
  )
}
```

## 3 — Structured Data: Breadcrumbs (בכל דף פנימי)

```tsx
// components/seo/BreadcrumbSchema.tsx
export function BreadcrumbSchema({ items }: { items: { name: string; url: string }[] }) {
  const schema = {
    "@context": "https://schema.org",
    "@type": "BreadcrumbList",
    itemListElement: items.map((item, i) => ({
      "@type": "ListItem",
      position: i + 1,
      name: item.name,
      item: item.url,
    })),
  }
  return <script type="application/ld+json" dangerouslySetInnerHTML={{ __html: JSON.stringify(schema) }} />
}
```

## 4 — Sitemap דינמי

```ts
// app/sitemap.ts
import { MetadataRoute } from "next"

const domain = process.env.NEXT_PUBLIC_DOMAIN!
const locales = ["he", "en"]

// Static pages — site_generator.py מחליף לפי הדפים שנמצאו ב-Site Archaeology
const staticPages = ["", "about", "services", "gallery", "booking", "contact", "accessibility-statement"]

export default function sitemap(): MetadataRoute.Sitemap {
  return locales.flatMap(locale =>
    staticPages.map(page => ({
      url: `https://${domain}/${locale}${page ? `/${page}` : ""}`,
      lastModified: new Date(),
      changeFrequency: page === "" ? "weekly" : "monthly" as const,
      priority: page === "" ? 1 : page === "services" ? 0.9 : 0.8,
    }))
  )
}
```

## 8 — Open Graph Image דינמי

```tsx
// app/og/route.tsx (Next.js ImageResponse)
import { ImageResponse } from "next/og"

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url)
  const title = searchParams.get("title") ?? process.env.NEXT_PUBLIC_BUSINESS_NAME!
  const subtitle = searchParams.get("subtitle") ?? ""
  
  return new ImageResponse(
    <div
      style={{
        background: "var(--color-surface)",
        width: "100%", height: "100%",
        display: "flex", flexDirection: "column",
        alignItems: "center", justifyContent: "center",
        fontFamily: "Heebo, sans-serif",
      }}
    >
      <h1 style={{ fontSize: 72, fontWeight: 800 }}>{title}</h1>
      {subtitle && <p style={{ fontSize: 36 }}>{subtitle}</p>}
    </div>,
    { width: 1200, height: 630 }
  )
}
```
