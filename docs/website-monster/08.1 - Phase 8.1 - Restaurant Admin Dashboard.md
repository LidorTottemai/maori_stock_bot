```typescript
// middleware.ts (root)
export const config = { matcher: ["/admin/:path*"] }

export default function middleware(req: NextRequest) {
  const token = req.cookies.get("admin_token")?.value
  if (!token && !req.nextUrl.pathname.startsWith("/admin/login")) {
    return NextResponse.redirect(new URL("/admin/login", req.url))
  }
  return NextResponse.next()
}
```

```typescript
// app/api/admin/login/route.ts
import { NextRequest, NextResponse } from "next/server"
import { SignJWT } from "jose"

export async function POST(req: NextRequest) {
  const { email, password } = await req.json()

  const res = await fetch(`${process.env.API_URL}/api/v1/auth/login`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ email, password }),
  })
  if (!res.ok) return NextResponse.json({ error: "אימייל או סיסמה שגויים" }, { status: 401 })

  const { place_id } = await res.json()
  const token = await new SignJWT({ place_id })
    .setProtectedHeader({ alg: "HS256" })
    .setExpirationTime("30d")
    .sign(new TextEncoder().encode(process.env.JWT_SECRET))

  const response = NextResponse.json({ ok: true })
  response.cookies.set("admin_token", token, { httpOnly: true, secure: true, maxAge: 60 * 60 * 24 * 30 })
  return response
}
```

## פרטי הזמנה + Status Pipeline

```tsx
// app/admin/orders/[id]/page.tsx
const NEXT_STATUS: Record<string, string | null> = {
  pending:   "confirmed",
  confirmed: "preparing",
  preparing: "ready",
  ready:     "completed",
  completed: null,
  cancelled: null,
}

const NEXT_LABEL: Record<string, string> = {
  pending:   "✅ אשר",
  confirmed: "🔄 התחל הכנה",
  preparing: "✅ מוכן",
  ready:     "🚗 הוגש / יצא",
}

function OrderActions({ order, onUpdate }: { order: any; onUpdate: () => void }) {
  const next = NEXT_STATUS[order.status]

  const advance = async () => {
    if (!next) return
    await fetch(`/api/admin/orders/${order.id}/status`, {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ status: next }),
    })
    onUpdate()
  }

  const cancel = async () => {
    if (!confirm("לבטל הזמנה?")) return
    await fetch(`/api/admin/orders/${order.id}/cancel`, { method: "POST" })
    onUpdate()
  }

  return (
    <div className="flex gap-2">
      {next && (
        <button onClick={advance}
                className="text-sm bg-[var(--color-primary)] text-white px-3 py-1 rounded-lg">
          {NEXT_LABEL[order.status]}
        </button>
      )}
      {order.status !== "cancelled" && order.status !== "completed" && (
        <button onClick={cancel} className="text-sm text-red-500 border border-red-200 px-3 py-1 rounded-lg">
          ביטול
        </button>
      )}
      <a href={`https://wa.me/${order.customer_phone}`} target="_blank"
         className="text-sm text-green-600 border border-green-200 px-3 py-1 rounded-lg">
        📱
      </a>
    </div>
  )
}
```

## VariantGroupEditor — עורך Variants

```tsx
// components/admin/VariantGroupEditor.tsx
// מאפשר לבעל מסעדה להגדיר קבוצות variants + options

interface VariantGroup {
  name:           string
  selection_type: "single_required" | "multi_optional" | "remove"
  is_required:    boolean
  options:        { label: string; price_delta: number }[]
}

export function VariantGroupEditor({ groups, onChange }: {
  groups:   VariantGroup[]
  onChange: (groups: VariantGroup[]) => void
}) {
  const addGroup = () => onChange([...groups, {
    name: "", selection_type: "single_required", is_required: false, options: [],
  }])

  const addOption = (gi: number) => {
    const updated = [...groups]
    updated[gi].options.push({ label: "", price_delta: 0 })
    onChange(updated)
  }

  return (
    <div dir="rtl" className="space-y-4">
      <div className="flex justify-between items-center">
        <h3 className="font-semibold">אפשרויות התאמה (variants)</h3>
        <button onClick={addGroup}
                className="text-sm bg-[var(--color-primary)] text-white px-3 py-1 rounded-lg">
          + הוסף קבוצה
        </button>
      </div>

      {groups.map((group, gi) => (
        <div key={gi} className="border border-[var(--color-border)] rounded-xl p-4 space-y-3">
          <div className="flex gap-3">
            <input
              value={group.name}
              onChange={e => { const u = [...groups]; u[gi].name = e.target.value; onChange(u) }}
              placeholder='שם קבוצה (למשל "גודל", "תוספות")'
              className="flex-1 border border-[var(--color-border)] rounded-lg px-3 py-1.5 text-sm"
            />
            <select
              value={group.selection_type}
              onChange={e => { const u = [...groups]; u[gi].selection_type = e.target.value as any; onChange(u) }}
              className="border border-[var(--color-border)] rounded-lg px-2 py-1.5 text-sm"
            >
              <option value="single_required">בחר אחד (חובה)</option>
              <option value="multi_optional">בחר כמה (רשות)</option>
              <option value="remove">הסרות</option>
            </select>
          </div>

          {/* Options */}
          {group.options.map((opt, oi) => (
            <div key={oi} className="flex gap-2 items-center">
              <input
                value={opt.label}
                onChange={e => { const u = [...groups]; u[gi].options[oi].label = e.target.value; onChange(u) }}
                placeholder={group.selection_type === "remove" ? "רכיב להסרה" : "תווית"}
                className="flex-1 border border-[var(--color-border)] rounded-lg px-3 py-1 text-sm"
              />
              <input
                type="number"
                value={opt.price_delta}
                onChange={e => { const u = [...groups]; u[gi].options[oi].price_delta = Number(e.target.value); onChange(u) }}
                placeholder="+₪"
                className="w-20 border border-[var(--color-border)] rounded-lg px-2 py-1 text-sm"
              />
              <button onClick={() => {
                const u = [...groups]; u[gi].options.splice(oi, 1); onChange(u)
              }} className="text-red-400 text-sm">✕</button>
            </div>
          ))}

          <button onClick={() => addOption(gi)}
                  className="text-sm text-[var(--color-primary)] hover:underline">
            + הוסף אפשרות
          </button>
        </div>
      ))}
    </div>
  )
}
```

## QR PDF Generator

```python
# app/services/qr_generator.py
import qrcode
import io
from reportlab.pdfgen import canvas
from reportlab.lib.units import cm

def generate_qr_pdf(place_id: str, table_count: int, domain: str) -> bytes:
    """מחזיר PDF עם QR Code לכל שולחן — להדפסה."""
    buf = io.BytesIO()
    c = canvas.Canvas(buf)

    for table in range(1, table_count + 1):
        url = f"https://{domain}/menu?table={table}"
        qr_img = qrcode.make(url)

        qr_buf = io.BytesIO()
        qr_img.save(qr_buf, format="PNG")
        qr_buf.seek(0)

        c.drawImage(qr_buf, 5*cm, 10*cm, width=10*cm, height=10*cm)
        c.setFont("Helvetica-Bold", 24)
        c.drawCentredString(10*cm, 9*cm, f"Table {table}")
        c.showPage()

    c.save()
    return buf.getvalue()
```
