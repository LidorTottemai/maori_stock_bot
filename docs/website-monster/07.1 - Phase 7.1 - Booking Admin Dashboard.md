## מה מקבל בעל העסק

```
/admin/                      → redirect → /admin/bookings
/admin/login/                ← התחברות
/admin/bookings/             ← לוח הזמנות (calendar ↔ table)
/admin/bookings/[id]/        ← פרטי הזמנה + פעולות
/admin/services/             ← ניהול שירותים (gallery ↔ table)
/admin/services/new/         ← הוסף שירות
/admin/services/[id]/edit/   ← עדכן שירות
/admin/settings/             ← שעות עבודה, ימי חופש, Telegram
```

```python
# app/api/v1/endpoints/admin_auth.py
from fastapi import APIRouter, Depends, HTTPException, Response
from pydantic import BaseModel
from sqlmodel import Session, select
from passlib.context import CryptContext
import jwt, os
from datetime import datetime, timedelta
from app.core.database import get_session
from app.models.dashboard_user import DashboardUser

router = APIRouter(prefix="/admin", tags=["admin"])
pwd_context = CryptContext(schemes=["bcrypt"])
SECRET = os.getenv("JWT_SECRET", "change-me-in-production")


class LoginBody(BaseModel):
    email:     str
    password:  str
    place_id:  str


@router.post("/login")
def admin_login(body: LoginBody, response: Response, session: Session = Depends(get_session)):
    user = session.exec(
        select(DashboardUser).where(DashboardUser.email == body.email)
    ).first()

    if not user or not pwd_context.verify(body.password, user.hashed_password):
        raise HTTPException(status_code=401, detail="פרטי התחברות שגויים")

    token = jwt.encode(
        {"sub": user.email, "place_id": body.place_id, "exp": datetime.utcnow() + timedelta(days=7)},
        SECRET,
        algorithm="HS256",
    )
    response.set_cookie("admin_token", token, httponly=True, secure=True, samesite="lax", max_age=604800)
    return {"ok": True, "place_id": body.place_id}


@router.post("/logout")
def admin_logout(response: Response):
    response.delete_cookie("admin_token")
    return {"ok": True}
```

### Next.js Middleware

```typescript
// middleware.ts
import { NextRequest, NextResponse } from "next/server"
import { jwtVerify } from "jose"

const SECRET = new TextEncoder().encode(process.env.JWT_SECRET ?? "change-me-in-production")

export async function middleware(req: NextRequest) {
  if (!req.nextUrl.pathname.startsWith("/admin")) return NextResponse.next()
  if (req.nextUrl.pathname === "/admin/login") return NextResponse.next()

  const token = req.cookies.get("admin_token")?.value
  if (!token) return NextResponse.redirect(new URL("/admin/login", req.url))

  try {
    await jwtVerify(token, SECRET)
    return NextResponse.next()
  } catch {
    return NextResponse.redirect(new URL("/admin/login", req.url))
  }
}

export const config = {
  matcher: ["/admin/:path*"],
}
```

### Login Page

```tsx
// app/[locale]/admin/login/page.tsx
"use client"
import { useState } from "react"
import { useRouter } from "next/navigation"

export default function AdminLogin() {
  const router = useRouter()
  const [email,    setEmail]    = useState("")
  const [password, setPassword] = useState("")
  const [error,    setError]    = useState("")
  const [loading,  setLoading]  = useState(false)

  const submit = async (e: React.FormEvent) => {
    e.preventDefault()
    setLoading(true)
    setError("")
    try {
      const res = await fetch("/api/admin/login", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ email, password, place_id: process.env.NEXT_PUBLIC_PLACE_ID }),
      })
      if (!res.ok) { setError("פרטי התחברות שגויים"); return }
      router.push("/admin/bookings")
    } finally {
      setLoading(false)
    }
  }

  return (
    <main dir="rtl" className="min-h-screen flex items-center justify-center p-8 bg-[var(--color-surface)]">
      <form onSubmit={submit} className="w-full max-w-sm space-y-5 bg-white rounded-3xl p-8 shadow-lg">
        <h1 className="text-2xl font-bold text-center">כניסה לניהול</h1>
        {error && <p className="text-red-500 text-sm text-center">{error}</p>}
        <div>
          <label className="block text-sm font-medium mb-1">אימייל</label>
          <input
            type="email"
            dir="ltr"
            value={email}
            onChange={e => setEmail(e.target.value)}
            required
            className="w-full border rounded-lg px-3 py-2.5"
          />
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">סיסמה</label>
          <input
            type="password"
            dir="ltr"
            value={password}
            onChange={e => setPassword(e.target.value)}
            required
            className="w-full border rounded-lg px-3 py-2.5"

        <button
          type="submit"
          disabled={loading}

        >
          {loading ? "מתחבר..." : "כניסה"}
        </button>
      </form>
    </main>
  )
}
```

```tsx
// app/[locale]/admin/services/page.tsx
"use client"
import { useEffect, useState } from "react"
import { ViewToggle } from "@tottemai/ui/booking"

      key: "is_active",

          onCheckedChange={checked => toggleActive(row.id, checked)}

      key: "id",
      label: "",

        <button
          onClick={() => router.push(`/admin/services/${id}/edit`)}

        >
          עריכה
        </button>
```

### Service Form (Add / Edit)

```tsx
// app/[locale]/admin/services/new/page.tsx (ו-[id]/edit/page.tsx)
"use client"
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import { Slider } from "@tottemai/ui/forms"
import { useRouter } from "next/navigation"

const schema = z.object({
  name:        z.string().min(2, "שם קצר מדי"),
  description: z.string().max(1000).optional(),
  duration:    z.number().min(15).max(360),
  price:       z.number().min(0),
  category:    z.string().optional(),
  is_active:   z.boolean().default(true),
})

type FormData = z.infer<typeof schema>

export default function ServiceForm({ initialData, serviceId }: { initialData?: Partial<FormData>; serviceId?: string }) {
  const router = useRouter()
  const form   = useForm<FormData>({
    resolver:      zodResolver(schema),
    defaultValues: initialData ?? { duration: 60, price: 0, is_active: true },
  })
  const duration = form.watch("duration")

  const submit = async (data: FormData) => {
    const url    = serviceId ? `/api/admin/services/${serviceId}` : "/api/admin/services"
    const method = serviceId ? "PUT" : "POST"
    await fetch(url, {
      method,
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(data),
    })
    router.push("/admin/services")
  }

  return (
    <form dir="rtl" onSubmit={form.handleSubmit(submit)} className="max-w-lg space-y-6 mx-auto p-6">
      <h1 className="text-2xl font-bold">{serviceId ? "עריכת שירות" : "שירות חדש"}</h1>

      <Field label="שם שירות *">
        <input {...form.register("name")} className="w-full border rounded-lg px-3 py-2.5" />
      </Field>

      <Field label="תיאור">
        <textarea {...form.register("description")} rows={4} className="w-full border rounded-lg px-3 py-2.5 resize-none" />
      </Field>

      <Field label={`משך — ${duration} דקות`}>
        <Slider
          min={15} max={180} step={15}
          value={[duration]}
          onValueChange={([v]) => form.setValue("duration", v)}
          className="mt-2"
        />
      </Field>

      <Field label="מחיר (₪)">
        <input
          type="number"
          min={0}
          {...form.register("price", { valueAsNumber: true })}
          className="w-full border rounded-lg px-3 py-2.5"
          dir="ltr"
        />
      </Field>

      <Field label="קטגוריה">
        <input {...form.register("category")} placeholder="עיסוי / פנים / שיער..." className="w-full border rounded-lg px-3 py-2.5" />
      </Field>

      <div className="flex gap-4">
        <button type="submit" className="flex-1 py-3 rounded-xl bg-[var(--color-primary)] text-white font-semibold">
          {serviceId ? "שמור שינויים" : "הוסף שירות"}
        </button>
        <button type="button" onClick={() => router.back()} className="flex-1 py-3 rounded-xl border font-semibold">
          ביטול
        </button>
      </div>
    </form>
  )
}

function Field({ label, children }: { label: string; children: React.ReactNode }) {
  return (
    <div className="space-y-1.5">
      <label className="block text-sm font-medium">{label}</label>
      {children}
    </div>
  )
}
```

## Business Settings

```tsx
// app/[locale]/admin/settings/page.tsx
"use client"
import { useEffect, useState } from "react"
import { Switch } from "@tottemai/ui/forms"

const DAYS = ["ראשון", "שני", "שלישי", "רביעי", "חמישי", "שישי", "שבת"]

interface DayHours { open: boolean; from: string; to: string }
type WorkingHours = Record<string, DayHours>

export default function AdminSettings() {
  const [hours,      setHours]      = useState<WorkingHours>({})
  const [telegramId, setTelegramId] = useState("")
  const [waPhone,    setWaPhone]    = useState("")
  const [saved,      setSaved]      = useState(false)

  useEffect(() => {
    fetch("/api/admin/settings").then(r => r.json()).then(d => {
      setHours(d.working_hours ?? {})
      setTelegramId(d.telegram_chat_id ?? "")
      setWaPhone(d.whatsapp_phone ?? "")
    })
  }, [])

  const updateDay = (i: number, patch: Partial<DayHours>) => {
    setHours(h => ({ ...h, [i]: { ...h[i], ...patch } }))
  }

  const save = async () => {
    await fetch("/api/admin/settings", {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ working_hours: hours, telegram_chat_id: telegramId, whatsapp_phone: waPhone }),
    })
    setSaved(true)
    setTimeout(() => setSaved(false), 2000)
  }

  return (
    <div dir="rtl" className="max-w-2xl mx-auto space-y-10 p-6">
      <h1 className="text-2xl font-bold">הגדרות עסק</h1>

      {/* שעות עבודה */}
      <section className="space-y-4">
        <h2 className="text-lg font-semibold">שעות פעילות</h2>
        {DAYS.map((day, i) => {
          const dh = hours[i] ?? { open: i !== 6, from: "09:00", to: "18:00" }
          return (
            <div key={i} className="flex items-center gap-4 flex-wrap">
              <Switch
                checked={dh.open}
                onCheckedChange={open => updateDay(i, { open })}
              />
              <span className="w-16 font-medium">{day}</span>
              {dh.open && (
                <>
                  <input type="time" value={dh.from} onChange={e => updateDay(i, { from: e.target.value })} className="border rounded px-2 py-1" />
                  <span className="text-[var(--color-text-muted)]">עד</span>
                  <input type="time" value={dh.to}   onChange={e => updateDay(i, { to: e.target.value })}   className="border rounded px-2 py-1" />
                </>
              )}
              {!dh.open && <span className="text-[var(--color-text-muted)] text-sm">סגור</span>}
            </div>
          )
        })}
      </section>

      {/* Telegram */}
      <section className="space-y-3">
        <h2 className="text-lg font-semibold">התראות</h2>
        <div>
          <label className="block text-sm font-medium mb-1">Telegram Chat ID</label>
          <input
            dir="ltr"
            value={telegramId}
            onChange={e => setTelegramId(e.target.value)}
            placeholder="-1001234567890"
            className="w-full border rounded-lg px-3 py-2.5"
          />
          <p className="text-xs text-[var(--color-text-muted)] mt-1">הוסף את הבוט לקבוצה וקבל את ה-chat ID</p>
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">WhatsApp לעסק</label>
          <input
            dir="ltr"
            value={waPhone}
            onChange={e => setWaPhone(e.target.value)}
            placeholder="972521234567"
            className="w-full border rounded-lg px-3 py-2.5"
          />
        </div>
      </section>

      <button
        onClick={save}
        className="w-full py-3 rounded-xl bg-[var(--color-primary)] text-white font-semibold"
      >
        {saved ? "✅ נשמר!" : "שמור הגדרות"}
      </button>
    </div>
  )
}
```
