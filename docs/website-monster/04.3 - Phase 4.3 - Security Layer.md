## 2 — Rate Limiting (middleware.ts)

```ts
// middleware.ts
import { NextRequest, NextResponse } from "next/server"

// In-memory rate limit — מספיק לאתרי עסקים קטנים (single-instance)
// לscale גדול יותר → Upstash Redis
const rateLimit = new Map<string, { count: number; reset: number }>()

const LIMITS: Record<string, { max: number; windowMs: number }> = {
  "/api/bookings": { max: 5,  windowMs: 60_000 },   // 5 הזמנות/דקה
  "/api/contact":  { max: 3,  windowMs: 60_000 },   // 3 פניות/דקה
  "/api/":         { max: 60, windowMs: 60_000 },   // כל שאר ה-API
}

function getLimit(pathname: string) {
  for (const [prefix, limit] of Object.entries(LIMITS)) {
    if (pathname.startsWith(prefix)) return limit
  }
  return null
}

export function middleware(req: NextRequest) {
  const pathname = req.nextUrl.pathname

  if (!pathname.startsWith("/api/")) return NextResponse.next()

  const limitConfig = getLimit(pathname)
  if (!limitConfig) return NextResponse.next()

  const ip = req.headers.get("x-forwarded-for")?.split(",")[0].trim() ?? "unknown"
  const key = `${ip}:${pathname}`
  const now = Date.now()
  const entry = rateLimit.get(key)

  if (entry && entry.reset > now) {
    if (entry.count >= limitConfig.max) {
      return NextResponse.json(
        { error: "יותר מדי בקשות. נסה שוב בעוד דקה." },
        {
          status: 429,
          headers: {
            "Retry-After": String(Math.ceil((entry.reset - now) / 1000)),
            "X-RateLimit-Limit": String(limitConfig.max),
            "X-RateLimit-Remaining": "0",
          },
        }
      )
    }
    entry.count++
  } else {
    rateLimit.set(key, { count: 1, reset: now + limitConfig.windowMs })
  }

  // ניקוי entries ישנים (כל 5 דקות)
  if (Math.random() < 0.01) {
    for (const [k, v] of rateLimit.entries()) {
      if (v.reset < now) rateLimit.delete(k)
    }
  }

  return NextResponse.next()
}

export const config = {
  matcher: ["/api/:path*"],
}
```

## 3 — Environment Variables Validation

```ts
// lib/env.ts — נטען פעם אחת ב-startup
import { z } from "zod"

const envSchema = z.object({
  // חובה לכל אתר
  NEXT_PUBLIC_BUSINESS_NAME:        z.string().min(2),
  NEXT_PUBLIC_DOMAIN:               z.string().min(3),
  NEXT_PUBLIC_PHONE:                z.string().regex(/^05\d{8}$/, "מספר טלפון ישראלי"),
  NEXT_PUBLIC_CITY:                 z.string().min(2),
  NEXT_PUBLIC_BUSINESS_DESCRIPTION: z.string().min(10),
  API_URL:                          z.string().url(),

  // אופציונלי — רק אם יש מערכת הזמנות
  CARDCOM_TERMINAL:    z.string().optional(),
  CARDCOM_USERNAME:    z.string().optional(),
  TELEGRAM_BOT_TOKEN:  z.string().optional(),
  TELEGRAM_CHAT_ID:    z.string().optional(),

  // אופציונלי — analytics
  GOOGLE_SITE_VERIFICATION: z.string().optional(),
  NEXT_PUBLIC_GA_ID:         z.string().optional(),
})

function validateEnv() {
  const result = envSchema.safeParse(process.env)
  if (!result.success) {
    const errors = result.error.flatten().fieldErrors
    console.error("❌ חסרים משתני סביבה:", JSON.stringify(errors, null, 2))
    throw new Error(`חסרים משתני סביבה חיוניים: ${Object.keys(errors).join(", ")}`)
  }
  return result.data
}

export const env = validateEnv()
```

```ts
      { status: 400 }
    )
  }
```

## 5 — CSRF Protection

```ts
// lib/csrf.ts — token-based CSRF לטפסים מורכבים
// Next.js App Router + Server Actions: CSRF מובנה דרך same-origin checks
// לAPI routes חיצוניים — כדאי:

export function verifyCsrf(req: NextRequest): boolean {
  const origin = req.headers.get("origin")
  const host   = req.headers.get("host")
  if (!origin || !host) return false
  return new URL(origin).host === host
}

// שימוש ב-POST routes:
if (!verifyCsrf(req)) {
  return NextResponse.json({ error: "Unauthorized" }, { status: 403 })
}
```

## 6 — Input Sanitization

```ts
// lib/sanitize.ts
// לטקסט חופשי שמוצג ב-HTML (הודעות, הערות)

export function sanitizeText(input: string): string {
  return input
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#x27;")
    .replace(/\//g, "&#x2F;")
    .trim()
}

// Zod transformer:
const safeString = z.string().transform(sanitizeText)
```
