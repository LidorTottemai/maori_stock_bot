### מה חולצים מכל אתר (DesignSignals)

```python
class DesignSignals(BaseModel):
    url: str

    # טיפוגרפיה
    fonts: list[str]              # font-family computed מ-h1, body, nav
    font_sizes: list[str]         # clamp() / vw sizes שנמצאו בCSS

    # צבעים
    bg_color: str                 # computed backgroundColor של body
    text_color: str               # computed color של body
    accent_colors: list[str]      # צבעים בולטים מbuttons/links/borders

    # ספריות אנימציה
    has_gsap: bool
    has_framer: bool              # framer-motion / motion.dev
    has_lenis: bool
    has_locomotive: bool          # locomotive-scroll
    has_three: bool               # three.js

    # מאפייני layout
    has_custom_cursor: bool
    has_horizontal_scroll: bool
    has_full_viewport_sections: bool
    has_parallax: bool
    has_sticky_nav: bool

    # מבנה דף
    nav_items: list[str]          # כותרות בnavbar
    section_count: int            # כמה sections בdף
    has_video_bg: bool
    has_canvas_bg: bool           # three.js / p5.js / canvas

    # איכות
    lighthouse_perf: int | None   # 0-100, אם אפשר לקבל
    is_mobile_responsive: bool
```

```python
class InspirationReport(BaseModel):
```

```python
# app/services/inspiration_crawler.py
import asyncio
import re
from playwright.async_api import async_playwright
import httpx
from pydantic import BaseModel
import logging

logger = logging.getLogger(__name__)

AWWWARDS_SEARCH = "https://www.awwwards.com/websites/?text={query}"
GOOGLE_SEARCH_TMPL = "best {category} website design award winning 2025"

CATEGORY_QUERIES = {
    "ספא": "spa beauty salon website",
    "מסעדה": "restaurant food website",
    "כושר": "fitness gym website",
    "קליניקה": "clinic healthcare website",
    "קמעונאות": "retail shop ecommerce website",
    "עורך דין": "law firm website",
    "נדלן": "real estate website",
    "אדריכלות": "architecture studio website",
    "אמנות": "art gallery artist website",
}


async def get_inspiration(category: str, http: httpx.AsyncClient) -> InspirationReport:
    """נקודת כניסה — מחזיר InspirationReport לקטגוריה."""
    query = CATEGORY_QUERIES.get(category, f"{category} website")

    # חיפוש במקביל
    urls = await _search_top_sites(query, http)
    
    if not urls:
        logger.warning(f"inspiration_crawler: no URLs found for '{category}', using fallback")
        return _fallback_report(category)

    # סרוק עד 5 אתרים במקביל
    tasks = [_extract_design_signals(url) for url in urls[:5]]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    signals = [r for r in results if isinstance(r, DesignSignals)]

    if not signals:
        return _fallback_report(category)

    return _synthesize(signals, category)


async def _search_top_sites(query: str, http: httpx.AsyncClient) -> list[str]:
    """חיפוש Awwwards + DuckDuckGo → רשימת URLs."""
    urls = []

    # Awwwards
    try:
        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            page = await browser.new_page()
            await page.goto(
                f"https://www.awwwards.com/websites/?text={query.replace(' ', '+')}",
                timeout=15000,
            )
            await page.wait_for_timeout(2000)
            links = await page.evaluate("""() =>
                Array.from(document.querySelectorAll('a[href*="/sites/"]'))
                    .map(a => a.href)
                    .filter(h => h.includes('awwwards.com/sites/'))
                    .slice(0, 8)
            """)
            # הוצא את הURL האמיתי של האתר (לא Awwwards)
            for link in links:
                site_url = await _get_real_url_from_awwwards(link, page)
                if site_url:
                    urls.append(site_url)
            await browser.close()
    except Exception as e:
        logger.warning(f"Awwwards search failed: {e}")

    # fallback — DuckDuckGo
    if len(urls) < 3:
        try:
            resp = await http.get(
                "https://api.duckduckgo.com/",
                params={"q": f"{query} award winning", "format": "json", "no_html": 1},
                timeout=10,
            )
            data = resp.json()
            for result in data.get("Results", [])[:5]:
                url = result.get("FirstURL", "")
                if url and not any(x in url for x in ["awwwards", "cssdesignawards", "google"]):
                    urls.append(url)
        except Exception as e:
            logger.warning(f"DuckDuckGo fallback failed: {e}")

    return list(dict.fromkeys(urls))  # dedup, maintain order


async def _extract_design_signals(url: str) -> DesignSignals:
    """Playwright: חולץ design signals מאתר אחד."""
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        try:
            page = await browser.new_page()
            await page.set_viewport_size({"width": 1440, "height": 900})
            await page.goto(url, timeout=20000, wait_until="domcontentloaded")
            await page.wait_for_timeout(3000)  # זמן לאנימציות להתחיל

            # גופנים
            fonts = await page.evaluate("""() => {
                const selectors = ['h1', 'h2', 'body', 'nav', 'p']
                const found = new Set()
                selectors.forEach(sel => {
                    const el = document.querySelector(sel)
                    if (el) found.add(
                        getComputedStyle(el).fontFamily.split(',')[0]
                            .replace(/['"]/g, '').trim()
                    )
                })
                return [...found].filter(f => f && f !== 'inherit')
            }""")

            # צבעים
            colors = await page.evaluate("""() => {
                const s = getComputedStyle(document.body)
                const header = document.querySelector('header, nav')
                const btn = document.querySelector('button, a.btn, .btn, [class*="button"]')
                const cols = [
                    s.backgroundColor,
                    s.color,
                    header ? getComputedStyle(header).backgroundColor : null,
                    btn ? getComputedStyle(btn).backgroundColor : null,
                    btn ? getComputedStyle(btn).color : null,
                ].filter(c => c && c !== 'rgba(0, 0, 0, 0)' && c !== 'transparent')
                return [...new Set(cols)].slice(0, 6)
            }""")

            # ספריות
            scripts = await page.evaluate("""() =>
                Array.from(document.querySelectorAll('script[src]')).map(s => s.src)
            """)
            inline_scripts = await page.evaluate("""() =>
                Array.from(document.querySelectorAll('script:not([src])')).map(s => s.textContent).join(' ')
            """)
            all_script_text = " ".join(scripts) + inline_scripts

            # CSS
            css_text = await page.evaluate("""() =>
                Array.from(document.styleSheets).flatMap(ss => {
                    try { return Array.from(ss.cssRules).map(r => r.cssText) }
                    catch { return [] }
                }).join(' ')
            """)

            # layout
            has_horizontal = await page.evaluate("""() =>
                !!document.querySelector('[class*="horizontal"], [class*="scroll-x"]') ||
                document.documentElement.scrollWidth > window.innerWidth * 1.5
            """)
            section_count = await page.evaluate("""() =>
                document.querySelectorAll('section, [class*="section"]').length
            """)
            has_video_bg = await page.evaluate("""() =>
                !!document.querySelector('video[autoplay], video[class*="bg"], [class*="video-bg"]')
            """)
            has_canvas = await page.evaluate("""() =>
                document.querySelectorAll('canvas').length > 0
            """)
            has_custom_cursor = await page.evaluate("""() =>
                !!document.querySelector('[class*="cursor"], [id*="cursor"]') ||
                document.querySelectorAll('[class*="cursor"]').length > 0
            """)

            return DesignSignals(
                url=url,
                fonts=fonts[:4],
                font_sizes=_extract_clamp_sizes(css_text),
                bg_color=colors[0] if colors else "",
                text_color=colors[1] if len(colors) > 1 else "",
                accent_colors=colors[2:],
                has_gsap="gsap" in all_script_text.lower(),
                has_framer="framer" in all_script_text.lower() or "motion" in all_script_text.lower(),
                has_lenis="lenis" in all_script_text.lower(),
                has_locomotive="locomotive" in all_script_text.lower(),
                has_three="three" in all_script_text.lower() or "webgl" in all_script_text.lower(),
                has_custom_cursor=has_custom_cursor,
                has_horizontal_scroll=has_horizontal,
                has_full_viewport_sections="100vh" in css_text or "h-screen" in css_text,
                has_parallax="parallax" in all_script_text.lower() or "scrolltrigger" in all_script_text.lower(),
                has_sticky_nav="sticky" in css_text or "position: fixed" in css_text,
                nav_items=[],
                section_count=section_count,
                has_video_bg=has_video_bg,
                has_canvas_bg=has_canvas,
                is_mobile_responsive="viewport" in await page.content(),
            )
        finally:
            await browser.close()


def _extract_clamp_sizes(css: str) -> list[str]:
    return re.findall(r"clamp\([^)]+\)", css)[:5]


def _synthesize(signals: list[DesignSignals], category: str) -> InspirationReport:
    """מסנתז רשימת DesignSignals ל-InspirationReport אחד."""
    from collections import Counter

    all_fonts = [f for s in signals for f in s.fonts]
    top_fonts = [f for f, _ in Counter(all_fonts).most_common(5)]

    animation_votes = {
        "GSAP + ScrollTrigger": sum(1 for s in signals if s.has_gsap),
        "Framer Motion / motion.dev": sum(1 for s in signals if s.has_framer),
        "Lenis smooth scroll": sum(1 for s in signals if s.has_lenis),
        "Locomotive Scroll": sum(1 for s in signals if s.has_locomotive),
        "Three.js / WebGL": sum(1 for s in signals if s.has_three),
    }
    animation_stack = [lib for lib, count in animation_votes.items() if count >= 2]

    patterns = []
    if sum(1 for s in signals if s.has_custom_cursor) >= 2:
        patterns.append("custom cursor")
    if sum(1 for s in signals if s.has_horizontal_scroll) >= 2:
        patterns.append("horizontal scroll gallery")
    if sum(1 for s in signals if s.has_full_viewport_sections) >= 3:
        patterns.append("full-viewport sections")
    if sum(1 for s in signals if s.has_parallax) >= 2:
        patterns.append("parallax scrolling")
    if sum(1 for s in signals if s.has_video_bg) >= 2:
        patterns.append("video hero background")
    if sum(1 for s in signals if s.has_canvas_bg) >= 2:
        patterns.append("WebGL / canvas background")

    avg_sections = sum(s.section_count for s in signals) / len(signals)

    dark_count = sum(1 for s in signals if _is_dark(s.bg_color))
    dominant = "dark background with vibrant accent" if dark_count > len(signals) / 2 else "light/minimal with strong typography"

    summary = _build_summary(category, top_fonts, animation_stack, patterns, dominant, avg_sections)

    return InspirationReport(
        category=category,
        sites_analyzed=len(signals),
        top_fonts=top_fonts,
        font_style=_describe_font_style(top_fonts),
        dominant_approach=dominant,
        common_accent_colors=[],
        animation_stack=animation_stack,
        animation_patterns=patterns,
        layout_patterns=patterns,
        avg_section_count=avg_sections,
        summary_for_claude=summary,
    )


def _is_dark(color: str) -> bool:
    """בדיקה גסה אם bg כהה."""
    if not color:
        return False
    nums = re.findall(r"\d+", color)
    if len(nums) >= 3:
        r, g, b = int(nums[0]), int(nums[1]), int(nums[2])
        return (r * 299 + g * 587 + b * 114) / 1000 < 128
    return False


def _describe_font_style(fonts: list[str]) -> str:
    serif_keywords = ["Serif", "Playfair", "Georgia", "Times", "Cormorant", "Lora"]
    has_serif = any(any(k in f for k in serif_keywords) for f in fonts)
    return "serif display + sans-serif body" if has_serif else "grotesque / geometric sans-serif"


def _build_summary(category, fonts, animations, patterns, dominant, sections) -> str:
    fonts_str = ", ".join(fonts[:3]) if fonts else "Syne, Inter"
    anim_str = ", ".join(animations[:3]) if animations else "GSAP, Framer Motion"
    patterns_str = ", ".join(patterns[:4]) if patterns else "full-viewport sections, text reveal"
    return f"""## INSPIRATION FROM WORLD-CLASS {category.upper()} WEBSITES

You analyzed {len(fonts)} top-tier award-winning {category} websites. Here's what sets them apart:

**Typography:** The best sites use {fonts_str}. NOT generic Inter. Choose a distinctive display font.
**Color approach:** {dominant}. Be bold, not safe.
**Animation stack:** {anim_str} — these are the tools the winners use.
**Layout patterns:** {patterns_str}.
**Sections:** Average {sections:.0f} focused sections — not one scrolling SPA dump.

Apply these insights. Your output should look like it belongs on Awwwards, not a WordPress theme directory.
"""


def _fallback_report(category: str) -> InspirationReport:
    return InspirationReport(
        category=category,
        sites_analyzed=0,
        top_fonts=["Syne", "Playfair Display", "DM Sans"],
        font_style="serif display + geometric sans",
        dominant_approach="dark with vibrant accent",
        common_accent_colors=[],
        animation_stack=["GSAP", "Framer Motion", "Lenis"],
        animation_patterns=["full-viewport sections", "text reveal", "custom cursor"],
        layout_patterns=["full-viewport sections", "parallax"],
        avg_section_count=6,
        summary_for_claude=f"Inspiration data unavailable for {category}. Default to bold, distinctive design.",
    )
```

```python
# app/services/rebuilder.py

from app.services.inspiration_crawler import get_inspiration
from app.services.competitor_researcher import research_competitors

async def rebuild_site(lead, settings, http):

    # שלב 1: מחקר מתחרים ישראלים (קיים)
    insights = await research_competitors(lead, http)

    # שלב 2: השראה מאתרי שורה ראשונה עולמיים (חדש)
    inspiration = await get_inspiration(lead.category, http)

    # שלב 3: מיזוג — מחקר + השראה
    combined_summary = (
        f"{insights.summary_for_claude}\n\n"
        f"{inspiration.summary_for_claude}"
    )

    # שלב 4: בנייה עם CLAUDE.md מועשר
    await run_site_generator(lead, combined_summary, settings)
```
