## רכיב 1 — כפתור הנגישות (AccessibilityWidget)

```tsx
// components/a11y/AccessibilityWidget.tsx
"use client"
import { useState, useEffect, useRef } from "react"
import { cn } from "@tottemai/ui"

interface A11yOptions {
  fontSize: number          // 0 = normal, 1 = +1 step, 2 = +2 steps
  contrast: "normal" | "high" | "inverted"
  highlightLinks: boolean
  underlineLinks: boolean
  bigCursor: boolean
  stopAnimations: boolean
  readableFont: boolean     // → Arial / system-ui
  lineHeight: number        // 0 = normal, 1 = 1.5x, 2 = 2x
  letterSpacing: number     // 0 = normal, 1 = +1px, 2 = +3px
  wordSpacing: number       // 0 = normal, 1 = wider
}

const DEFAULT: A11yOptions = {
  fontSize: 0, contrast: "normal", highlightLinks: false,
  underlineLinks: false, bigCursor: false, stopAnimations: false,
  readableFont: false, lineHeight: 0, letterSpacing: 0, wordSpacing: 0,
}

export function AccessibilityWidget({ statementUrl = "/accessibility-statement" }: { statementUrl?: string }) {
  const [open, setOpen] = useState(false)
  const [opts, setOpts] = useState<A11yOptions>(DEFAULT)
  const panelRef = useRef<HTMLDivElement>(null)

  // Apply options to <html> data attributes → CSS picks them up
  useEffect(() => {
    const html = document.documentElement
    html.dataset.a11yFontSize     = String(opts.fontSize)
    html.dataset.a11yContrast     = opts.contrast
    html.dataset.a11yHighlight    = String(opts.highlightLinks)
    html.dataset.a11yUnderline    = String(opts.underlineLinks)
    html.dataset.a11yCursor       = String(opts.bigCursor)
    html.dataset.a11yAnimations   = String(opts.stopAnimations)
    html.dataset.a11yFont         = String(opts.readableFont)
    html.dataset.a11yLineHeight   = String(opts.lineHeight)
    html.dataset.a11yLetterSpace  = String(opts.letterSpacing)
    html.dataset.a11yWordSpace    = String(opts.wordSpacing)
    localStorage.setItem("a11y", JSON.stringify(opts))
  }, [opts])

  // Restore from localStorage
  useEffect(() => {
    const saved = localStorage.getItem("a11y")
    if (saved) setOpts(JSON.parse(saved))
  }, [])

  // Close on outside click / Escape
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => { if (e.key === "Escape") setOpen(false) }
    const onClick = (e: MouseEvent) => {
      if (panelRef.current && !panelRef.current.contains(e.target as Node)) setOpen(false)
    }
    document.addEventListener("keydown", onKey)
    document.addEventListener("mousedown", onClick)
    return () => { document.removeEventListener("keydown", onKey); document.removeEventListener("mousedown", onClick) }
  }, [])

  const toggle = (key: keyof A11yOptions, type: "bool" | "cycle3") => {
    setOpts(prev => {
      if (type === "bool") return { ...prev, [key]: !prev[key as keyof A11yOptions] }
      const cur = prev[key] as number
      return { ...prev, [key]: (cur + 1) % 3 }
    })
  }

  return (
    <div ref={panelRef} className="fixed bottom-6 start-6 z-[9999]" dir="rtl">
      {/* הכפתור הצף — תמיד נראה */}
      <button
        onClick={() => setOpen(v => !v)}
        aria-label="פתח תפריט נגישות"
        aria-expanded={open}
        aria-haspopup="dialog"
        className={cn(
          "w-14 h-14 rounded-full shadow-xl",
          "bg-[var(--color-primary)] text-white",
          "flex items-center justify-center",
          "focus:outline-none focus:ring-4 focus:ring-[var(--color-primary)] focus:ring-offset-2",
          "transition-transform hover:scale-110",
        )}
      >
        {/* wheelchair SVG icon */}
        <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" aria-hidden="true">
          <circle cx="12" cy="4" r="2"/>
          <path d="M10 9L8 17h8l1-5H10V9z"/>
          <path d="M8 17c0 2.2 1.8 4 4 4s4-1.8 4-4"/>
        </svg>
      </button>

      {/* פאנל הנגישות */}
      {open && (
        <div
          role="dialog"
          aria-modal="true"
          aria-label="תפריט נגישות"
          className={cn(
            "absolute bottom-16 start-0",
            "w-72 rounded-2xl shadow-2xl",
            "bg-[var(--color-surface)] border border-[var(--color-border)]",
            "p-4 space-y-3",
          )}
        >
          <h2 className="text-lg font-bold text-[var(--color-text)]">כלי נגישות</h2>

          {/* גודל גופן */}
          <A11yButton
            label="הגדלת טקסט"
            value={["רגיל", "גדול", "גדול מאוד"][opts.fontSize]}
            onClick={() => toggle("fontSize", "cycle3")}
            active={opts.fontSize > 0}
          />

          {/* ניגודיות */}
          <A11yButton
            label="ניגודיות גבוהה"
            value={opts.contrast === "normal" ? "כבוי" : opts.contrast === "high" ? "גבוהה" : "הפוכה"}
            onClick={() => setOpts(p => ({
              ...p,
              contrast: p.contrast === "normal" ? "high" : p.contrast === "high" ? "inverted" : "normal"
            }))}
            active={opts.contrast !== "normal"}
          />

          {/* הדגשת קישורים */}
          <A11yButton label="הדגש קישורים"   value={opts.highlightLinks ? "פעיל" : "כבוי"} onClick={() => toggle("highlightLinks",  "bool")} active={opts.highlightLinks} />
          <A11yButton label="קו תחת קישורים" value={opts.underlineLinks  ? "פעיל" : "כבוי"} onClick={() => toggle("underlineLinks",   "bool")} active={opts.underlineLinks} />

          {/* סמן גדול */}
          <A11yButton label="סמן גדול"        value={opts.bigCursor       ? "פעיל" : "כבוי"} onClick={() => toggle("bigCursor",        "bool")} active={opts.bigCursor} />

          {/* עצור אנימציות */}
          <A11yButton label="עצור אנימציות"  value={opts.stopAnimations   ? "פעיל" : "כבוי"} onClick={() => toggle("stopAnimations",   "bool")} active={opts.stopAnimations} />

          {/* גופן קריא */}
          <A11yButton label="גופן קריא"       value={opts.readableFont     ? "פעיל" : "כבוי"} onClick={() => toggle("readableFont",     "bool")} active={opts.readableFont} />

          {/* גובה שורה */}
          <A11yButton label="גובה שורה"       value={["רגיל","1.5x","2x"][opts.lineHeight]}   onClick={() => toggle("lineHeight",       "cycle3")} active={opts.lineHeight > 0} />

          {/* רווח אותיות */}
          <A11yButton label="רווח אותיות"    value={["רגיל","+1px","+3px"][opts.letterSpacing]} onClick={() => toggle("letterSpacing", "cycle3")} active={opts.letterSpacing > 0} />

          {/* איפוס */}
          <button
            onClick={() => setOpts(DEFAULT)}
            className="w-full text-center text-sm text-[var(--color-text-muted)] underline mt-2"
          >
            איפוס הגדרות
          </button>

          {/* קישור להצהרת נגישות */}
          <a
            href={statementUrl}
            className="block text-center text-sm text-[var(--color-primary)] underline"
          >
            הצהרת נגישות
          </a>
        </div>
      )}
    </div>
  )
}

function A11yButton({ label, value, onClick, active }: {
  label: string; value: string; onClick: () => void; active: boolean
}) {
  return (
    <button
      onClick={onClick}
      className={cn(
        "w-full flex justify-between items-center",
        "rounded-lg px-3 py-2 text-sm",
        "transition-colors",
        active
          ? "bg-[var(--color-primary)] text-white"
          : "bg-[var(--color-muted,#f3f4f6)] text-[var(--color-text)] hover:bg-[var(--color-border)]",
      )}
    >
      <span>{label}</span>
      <span className="text-xs opacity-70">{value}</span>
    </button>
  )
}
```

## מה נדחף ל-CLAUDE.md (מ-site_generator.py)

```
## ACCESSIBILITY (תקן ישראלי 5568 — חובה חוקית)

⚠️ קנס עד ₪50,000 לאתר ללא נגישות.

### חובה בכל אתר:
□ SkipLink — ראשון בDOM, לפני כל תוכן אחר
□ AccessibilityWidget — כפתור צף קבוע, אחרון בDOM
□ /accessibility-statement — דף מלא בעברית ואנגלית
□ <html lang="he" dir="rtl"> (או lang="en" dir="ltr")
□ id="main-content" על <main>

### WCAG 2.1 AA:
□ כל <img> חייב alt="" (דקורטיבי) או alt="תיאור"
□ כל interactive element — aria-label בעברית
□ Color contrast: טקסט ≥ 4.5:1, UI ≥ 3:1
□ Font sizes: rem בלבד. body minimum 1rem
□ Forms: <label> לכל <input>, error ב-aria-describedby
□ Modals: focus trap + Escape לסגירה + aria-modal="true"
□ Keyboard: Tab/Shift+Tab, Enter, Space, חצים בתפריטים
□ Focus outline: 2px solid var(--color-primary), offset: 2px
□ prefers-reduced-motion: כל אנימציה עטופה

### קבצים שחייבים להיות:
- components/a11y/SkipLink.tsx
- components/a11y/AccessibilityWidget.tsx
- app/[locale]/accessibility-statement/page.tsx
```

## זיהוי אוטומטי ב-dig() (Phase 4)

```python
# מ-site_archaeology.py
WIDGET_SELECTORS = [
    ".equalweb-btn", "#INDmenu-btn", ".userway-btn",
    "[data-userway]", ".adally-btn", "#nagich", ".ally-btn",
    "[id*='accessibility']", "[class*='accessibility-widget']",
    "[class*='a11y-btn']",
]

async def _detect_accessibility_widget(page) -> bool:
    for sel in WIDGET_SELECTORS:
        el = await page.query_selector(sel)
        if el:
            return True
    return False
```
